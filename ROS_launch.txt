Создаем launch-файл в директории пакета
Вызывается roslaunch "полный путь до пакета" or 
           roslaunch "имя пакета" "имя launch-файла"
Форма:
<launch>
/*******Базовый запуск*******/
<node pkg="|имя пакета|" type="|имя ноды|" name="|имя для запуска вместо default|"/>
/*******С выводом на экран*******/
<node pkg="" type="" name="" output="screen"/>
/*******С переназначением топика*******/
<node pkg="" type="" name="">  
<remap from="|изначальный топик|" to="|новый топик|"/>
</node>
/*******С передачей значения*******/
<arg name="|имя местной переменной|" default="|дефолтное значение|"/>  //Для задания дефолтного значения
<node pkg="" type="" name="">
<param name="имя переменной" type="|тип переменной(можно не указывать)|" value="|константное значение|"or"$(arg |имя местной переменной|)"/> 
//Назначается в value или константное значение, или переменная	
</node>   
/**Не забыть добавить в код ноды: ros::param::param<"тип переменной">("~|имя переменной|", "val куда записать", "значение по умолчанию")***/

</launch>
