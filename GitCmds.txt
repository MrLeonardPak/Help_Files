#Git_Commands

git init = в текущей папке создает инициализирует git-репозиторий и .git
/OR/
git clone <ssh> <имя папки> = загрузка репозитория из GitHub в текущее местоположение в <имя папки> 

git status = состояние текущей папки

git add = добавление папок и файлов текущей папки в локальный репозиторий (. = из текущей директории, -A = из всей рабочей области)

git commit -m "<комметнарий>" = подготовка текущей папки к загрузке

git push = выгрузка текущещей папки на GitHub

git pull = обновление папки
